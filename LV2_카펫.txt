#include <string>
#include <vector>

using namespace std;

vector<pair<int,int>> solution(int brown, int yellow) {
    vector<pair<int,int>> answer;
    bool breaking = false;
	int sum = brown + yellow;

	for (int i = 1; i <= sum/2; i++)
	{
		for (int j = 1; j <= sum/2; j++)
		{
			int tmp = 0;
			if (i >= j && i*j == sum)
			{
				for (int k = 1; k <= sum/2; k++)
				{
					
					tmp += ((i - (k * 2)) * 2 + (j - (k * 2)) * 2)+4;
					if (tmp == brown)
					{
						breaking = true;
						answer.push_back({ i,j });
						break;
					}
				}
			}
			if (breaking == true)
				break;
		}
		if (breaking == true)
			break;

	}
    return answer;
}
