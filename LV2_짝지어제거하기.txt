#include <iostream>
#include <string>
#include <stack>
using namespace std;

int solution(string s)
{
    
    stack<char> stk;

	for (unsigned int i = 0; i < s.length(); i++)
	{
		if (!stk.empty() && stk.top() == s[i])
		{
			stk.pop();
		}
		else
		{
			stk.push(s[i]);
		}
	}

	if (stk.empty())
		return 1;
	else
		return 0;

}
